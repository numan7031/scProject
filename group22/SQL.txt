CREATE TABLE user
(
  userID INT NOT NULL,
  fname VARCHAR(25) NOT NULL,
  lname VARCHAR(25) NOT NULL,
  email VARCHAR(25) NOT NULL,
  username VARCHAR(25) NOT NULL,
  password VARCHAR(25) NOT NULL,
  status VARCHAR(10) NOT NULL,
  image VARCHAR(25) NOT NULL,
  PRIMARY KEY (userID)
);

CREATE TABLE attractions
(
  attracID INT NOT NULL,
  atname VARCHAR(50) NOT NULL,
  adress VARCHAR(100) NOT NULL,
  toilet VARCHAR(25) NOT NULL,
  threeGfourG VARCHAR(25) NOT NULL,
  unseen VARCHAR(25) NOT NULL,
  traveladvice VARCHAR(200) NOT NULL,
  price INT NOT NULL,
  security VARCHAR(25) NOT NULL,
  facilitiesfordisabled VARCHAR(25) NOT NULL,
  advicefordisabled VARCHAR(200) NOT NULL,
  wifi VARCHAR(25) NOT NULL,
  history VARCHAR(500) NOT NULL,
  tourdesk VARCHAR(25) NOT NULL,
  lat VARCHAR(50) NOT NULL,
  long VARCHAR(50) NOT NULL,
  festival VARCHAR(25) NOT NULL,
  variousnature VARCHAR(25) NOT NULL,
  replacein VARCHAR(25) NOT NULL,
  replaceout VARCHAR(25) NOT NULL,
  PRIMARY KEY (attracID)
);

CREATE TABLE nearbyattractions
(
  nearID INT NOT NULL,
  atname VARCHAR(50) NOT NULL,
  attracID INT NOT NULL,
  PRIMARY KEY (nearID),
  FOREIGN KEY (attracID) REFERENCES attractions(attracID)
);

CREATE TABLE accommodation
(
  acID INT NOT NULL,
  acname VARCHAR(25) NOT NULL,
  lat VARCHAR(25) NOT NULL,
  long VARCHAR(25) NOT NULL,
  adress VARCHAR(100) NOT NULL,
  phone VARCHAR(11) NOT NULL,
  description VARCHAR(200) NOT NULL,
  attracID INT NOT NULL,
  PRIMARY KEY (acID),
  FOREIGN KEY (attracID) REFERENCES attractions(attracID)
);

CREATE TABLE restaurant
(
  resID INT NOT NULL,
  resname VARCHAR NOT NULL,
  lat INT NOT NULL,
  long INT NOT NULL,
  description INT NOT NULL,
  adress INT NOT NULL,
  phone INT NOT NULL,
  attracID INT NOT NULL,
  PRIMARY KEY (resID),
  FOREIGN KEY (attracID) REFERENCES attractions(attracID)
);

CREATE TABLE attracImages
(
  imageID INT NOT NULL,
  imageURL VARCHAR(200) NOT NULL,
  attracID INT NOT NULL,
  PRIMARY KEY (imageID),
  FOREIGN KEY (attracID) REFERENCES attractions(attracID)
);

CREATE TABLE accomImages
(
  imageID INT NOT NULL,
  imageURL VARCHAR(200) NOT NULL,
  acID INT NOT NULL,
  PRIMARY KEY (imageID),
  FOREIGN KEY (acID) REFERENCES accommodation(acID)
);

CREATE TABLE resImages
(
  imageID INT NOT NULL,
  imageURL VARCHAR(200) NOT NULL,
  resID INT NOT NULL,
  PRIMARY KEY (imageID),
  FOREIGN KEY (resID) REFERENCES restaurant(resID)
);

CREATE TABLE review
(
  revID INT NOT NULL,
  topic VARCHAR(25) NOT NULL,
  reviewdes VARCHAR(300) NOT NULL,
  score INT NOT NULL,
  userID INT NOT NULL,
  attracID INT NOT NULL,
  PRIMARY KEY (revID),
  FOREIGN KEY (userID) REFERENCES user(userID),
  FOREIGN KEY (attracID) REFERENCES attractions(attracID)
);

CREATE TABLE souvenir
(
  servID INT NOT NULL,
  name VARCHAR(25) NOT NULL,
  adress VARCHAR(200) NOT NULL,
  phone VARCHAR(11) NOT NULL,
  lat VARCHAR(25) NOT NULL,
  long VARCHAR(25) NOT NULL,
  attracID INT NOT NULL,
  PRIMARY KEY (servID),
  FOREIGN KEY (attracID) REFERENCES attractions(attracID)
);
